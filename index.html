Skype::Any

Skype API wrapper for Perl

use strict;
use warnings;
use Skype::Any;

my $skype = Skype::Any->new();
$skype->message_received(sub {
    my ($msg) = @_;
    my $body = $msg->body;
    if ($body eq 'ping') {
        $msg->chat->send_message('pong');
    }
});
$skype->run;
