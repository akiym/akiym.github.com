初めて書いたプログラム

use strict;
use warnings;
use Google::Search;
use LWP::UserAgent;
use File::Basename;

my %opt = ( dir => '.' );
Getopt::Long::GetOptions( \%opt, 'dir=s', 'help' );

my $query = shift;
mkdir $opt{dir} unless -e $opt{dir}; # no such directory

my $search = Google::Search->Image( q => $query );
my $ua = LWP::UserAgent->new( agent => 'Mozilla/5.0' );

my $result = $search->first;
while ($result) {
    my $filename = File::Basename::basename($result->uri);
    unless ($filename =~ /\.[^\.]+$/) { # no extension
         $filename .= '.jpg';
    }

    my $filepath = Path::Class::file($opt{dir}, $filename);

    print $result->number, " ", $result->uri, "\n";
    my $response = $ua->get( $result->uri,
                             ':content_file' => $filepath->stringify );
    unless ($response->is_success) {
        print "Failed to download...\n";
    }

    $result = $result->next;
}
